FORM display_data .
  LOOP AT git_final INTO gwa_final.
    WRITE: / gwa_final-matnr UNDER 'MATNR',
             " and other fields but need the below HOTSPOT on one.
             gwa_final-werks UNDER 'WEKRS' HOTSPOT,
    CLEAR gwa_final.
  ENDLOOP.
  ULINE.

" <MOST IMPORTANT FOR HANDLING_LINE_SELECTION>
FORM handle_line_selection.

" Define the data required in data include program like structs, internal
" tables, work area and GET CURSORS as well, still added get cursor.
  
" <add in data declarations include program>
  DATA: fname TYPE char30,
        fval  TYPE char30.

  GET CURSOR FIELD fname VALUE fval.

  IF fname = 'GWA_FINAL-WERKS'.
    SELECT matnr
           " and other fields
      INTO TABLE it_plant
      FROM mard
      WHERE werks = fval.

    IF sy-subrc IS INITIAL.
      WRITE: / 'MATNR' COLOR 1,
            " and others.
      ULINE.

      LOOP AT it_plant INTO wa_plant.
        WRITE: / wa_plant-matnr UNDER 'MATNR',
                " and others.
        CLEAR wa_plant.
      ENDLOOP.
      ULINE.
    ENDIF.
  ELSE.
    MESSAGE: 'PLEASE CLICK ON A VALUE IN THE "WERKS" FIELD.' TYPE 'I'.
  ENDIF.

ENDFORM.

" FieldCat Merge to Populate the first list
FORM populate_first_list_fcat .
  CALL FUNCTION 'REUSE_ALV_FIELDCATALOG_MERGE'
   EXPORTING
     I_PROGRAM_NAME               = sy-repid
     I_INTERNAL_TABNAME           = 'GIT_MTDC'
     I_STRUCTURE_NAME             = 'ZM234_MTDC'
    CHANGING
      ct_fieldcat                 = git_fcat[]
   EXCEPTIONS
     INCONSISTENT_INTERFACE       = 1
     PROGRAM_ERROR                = 2
     OTHERS                       = 3.
  IF sy-subrc IS INITIAL. " THIS IS THE IMPORTANT PART
    CLEAR gwa_fcat.
    LOOP AT git_fcat INTO gwa_fcat.
      IF gwa_fcat-fieldname = 'MATNR'. " Field Name to be changed as needed
        gwa_fcat-hotspot = 'X'.
        MODIFY git_fcat FROM gwa_fcat.
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDFORM.

" Upon LINE SELECTION (Basically GET CURSOR Method for ALVI)
FORM at_line_selection USING r_ucomm LIKE sy-ucomm
                             rs_selfield TYPE slis_selfield.

  CASE r_ucomm.
    WHEN '&IC1'.
      READ TABLE git_mtdc INDEX rs_selfield-tabindex INTO gwa_mtdc. " Get Cursor for ALV
      CHECK sy-subrc IS INITIAL.
      IF rs_selfield-sel_tab_field = 'GIT_MTDC-MBLNR'.
        SELECT  MBLNR
                " and other fields
           FROM ZM234_MTDS INTO TABLE git_mtds WHERE mblnr = rs_selfield-value.
        IF sy-subrc IS INITIAL.
          PERFORM display_mtds.
        ENDIF.
      ENDIF.
  ENDCASE.
ENDFORM.

  METHOD constructor.

     go_main->fetch_s1(
       EXPORTING

       CHANGING
