  PARAMETERS p_add RADIOBUTTON GROUP rg_1.
  PARAMETERS p_sub RADIOBUTTON GROUP rg_1.

  SELECTION-SCREEN: BEGIN OF LINE.
    SELECTION-SCREEN: COMMENT 11(10) text-003 FOR FIELD p_mul.
    SELECTION-SCREEN: POSITION 31.
    PARAMETERS p_mul RADIOBUTTON GROUP rg_1.

    SELECTION-SCREEN: COMMENT 35(10) text-004 FOR FIELD p_div.
    SELECTION-SCREEN: POSITION 45.
    PARAMETERS p_div RADIOBUTTON GROUP rg_1 DEFAULT 'X'.
  SELECTION-SCREEN: END OF LINE.
SELECTION-SCREEN: END OF BLOCK b0.

SELECTION-SCREEN: BEGIN OF BLOCK b1 WITH FRAME TITLE text-200.
  PARAMETERS: p_num1 TYPE char4 OBLIGATORY,
              p_num2 TYPE char4 OBLIGATORY.
SELECTION-SCREEN: END OF BLOCK b1.
SELECTION-SCREEN: END OF BLOCK b.



IF p_add = abap_true.
  p_result = p_num1 + p_num2.

ELSEIF p_sub = abap_true.
  IF p_num1 > p_num2.
    p_result = p_num1 - p_num2.
  ELSEIF p_num2 > p_num1.
    p_result = p_num2 - p_num1.
  ENDIF.

ELSEIF p_mul = abap_true.
  IF p_num1 = 0 OR p_num2 = 0.
    WRITE text-035.
  ELSE.
    p_result = p_num1 * p_num2.
  ENDIF.

ELSEIF p_div = abap_true.
  IF p_num1 = 0 OR p_num2 = 0.
    WRITE text-036.
  ELSEIF p_num1 >< 0 OR p_num2 >< 0.
    p_result = p_num1 / p_num2.
  ENDIF.

ENDIF.

WRITE p_result.
